podman network create chatapp-network

podman run \
       --name chatapp-postgres \
       -e POSTGRES_USER='postgres' -e POSTGRES_DB='chatapp' -e POSTGRES_PASSWORD='postgres' \
       -d --rm \
       -p 5432:5432 \
       --network chatapp-network \
       postgres

podman build -t chatapp-backend .

podman run \
       --name chatapp-backend \
       -d --rm \
       --network chatapp-network \
       -p 8000:8000 \
       chatapp-backend

podman build -t chatapp-frontend .

podman run \
       --name chatapp-frontend \
       -d \
       -p 5173:5173 \
       chatapp-frontend
